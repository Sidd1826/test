import json
import os
import requests
import math
import pandas as pd
import time
import urllib3
import concurrent.futures
from crypto_utils import encrypt_data, decrypt_data
from datetime import datetime, timedelta
from requests.adapters import HTTPAdapter
from urllib3.util.retry import Retry
from threading import Lock

# Disable SSL warnings
urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

# Token cache
_token_cache = {
    'token': None,
    'expires_at': None
}
_token_lock = Lock()

# Session
_session = None
_session_lock = Lock()

def get_session_with_retries():
    """Create a requests session with retry logic and connection pooling"""
    global _session
    
    if _session is not None:
        return _session
    
    with _session_lock:
        if _session is not None:
            return _session
        
        session = requests.Session()
        
        retry_strategy = Retry(
            total=3,
            backoff_factor=1,
            status_forcelist=[429, 500, 502, 503, 504],
            allowed_methods=["POST", "GET"]
        )
        
        adapter = HTTPAdapter(
            max_retries=retry_strategy,
            pool_connections=20,  # Increased for parallel
            pool_maxsize=20,
            pool_block=False
        )
        
        session.mount("http://", adapter)
        session.mount("https://", adapter)
        
        _session = session
        return session


def fri_auth(key_id: str, force_refresh=False) -> str:
    """Authenticate with FRI API and cache the token (thread-safe)"""
    global _token_cache
    
    with _token_lock:
        # Return cached token if valid
        if not force_refresh and _token_cache['token'] and _token_cache['expires_at']:
            if datetime.now() < _token_cache['expires_at']:
                return _token_cache['token']
        
        print("Fetching new authentication token...")
        
        payload = {
            "email": os.getenv("FRI_EMAIL"),
            "password": os.getenv("FRI_PASSWORD")
        }
        
        if not payload["email"] or not payload["password"]:
            raise ValueError("FRI_EMAIL and FRI_PASSWORD environment variables must be set")
        
        encrypted_payload = encrypt_data(payload, key_id)
        
        headers = {
            "Content-Type": "application/json"
        }
        
        AUTH_URL = "https://fritestgateway.sancharsaathi.gov.in/fri/v1/auth"
        
        session = get_session_with_retries()
        
        try:
            response = session.post(
                AUTH_URL,
                json=encrypted_payload,
                headers=headers,
                timeout=30,
                verify=False
            )
            
            response.raise_for_status()
            
            if not response.text:
                raise ValueError("Empty response from auth API")
            
            res = response.json()
            token = res.get('token')
            
            if not token:
                raise ValueError(f"No token in response. Response: {res}")
            
            # Cache the token
            _token_cache['token'] = token
            _token_cache['expires_at'] = datetime.now() + timedelta(minutes=120)
            
            print("✓ Authentication successful")
            return token
            
        except Exception as e:
            print(f"✗ Auth API Error: {e}")
            raise


def fri_count(key_id: str, count_date: str, max_retries=3) -> dict:
    """Get count of records for a specific date"""
    session = get_session_with_retries()
    
    for attempt in range(max_retries):
        try:
            token = fri_auth(key_id, force_refresh=(attempt > 0))
            
            headers = {
                "Content-Type": "application/json",
                "Authorization": token
            }
            
            body = {"date": count_date}
            FRI_COUNT_URL = "https://fritestgateway.sancharsaathi.gov.in/fri/v1/count"
            
            response = session.post(
                FRI_COUNT_URL,
                json=body,
                headers=headers,
                timeout=30,
                verify=False
            )
            
            if response.status_code == 401 and attempt < max_retries - 1:
                print("Token expired, refreshing...")
                with _token_lock:
                    _token_cache['token'] = None
                    _token_cache['expires_at'] = None
                time.sleep(1)
                continue
            
            response.raise_for_status()
            
            if not response.text:
                raise ValueError(f"Empty response from count API")
            
            result = response.json()
            print(f"✓ Count retrieved: {result.get('count', 0)} records")
            return result
        
        except requests.exceptions.ConnectionError as e:
            if attempt < max_retries - 1:
                wait_time = (2 ** attempt)
                print(f"Connection error, retrying in {wait_time}s...")
                time.sleep(wait_time)
            else:
                raise
        except Exception as e:
            print(f"✗ Count API Error: {e}")
            raise
    
    raise ValueError(f"Failed to get count after {max_retries} attempts")


def fri_data(key_id: str, count: int, offset: int, date: str, client_id: str, max_retries=3) -> dict:
    """Fetch FRI data with pagination (optimized for parallel calls)"""
    session = get_session_with_retries()
    
    for attempt in range(max_retries):
        try:
            token = fri_auth(key_id)
            
            headers = {
                "Content-Type": "application/json",
                "Authorization": token
            }
            
            body = {
                "date": date,
                "client_id": client_id,
                "offset": offset,
                "count": count
            }
            
            FRI_DATA_URL = "https://fritestgateway.sancharsaathi.gov.in/fri/v1/data"
            
            response = session.post(
                FRI_DATA_URL,
                json=body,
                headers=headers,
                timeout=45,  # Reduced timeout
                verify=False
            )
            
            if response.status_code == 401 and attempt < max_retries - 1:
                with _token_lock:
                    _token_cache['token'] = None
                    _token_cache['expires_at'] = None
                time.sleep(1)
                continue
            
            response.raise_for_status()
            
            if not response.text:
                raise ValueError("Empty response from data API")
            
            return response.json()
        
        except requests.exceptions.ConnectionError as e:
            if attempt < max_retries - 1:
                wait_time = min(2 ** attempt, 4)  # Cap at 4 seconds
                time.sleep(wait_time)
            else:
                raise
        except Exception as e:
            if attempt < max_retries - 1:
                time.sleep(1)
            else:
                raise
    
    raise ValueError(f"Failed to fetch data after {max_retries} attempts")


def fetch_and_decrypt_all(key_id: str, total_count: int, date: str, client_id: str, BATCH_SIZE: int = 2000, max_workers: int = 5) -> list:
    """
    Fetch and decrypt all data in PARALLEL batches
    
    Args:
        BATCH_SIZE: Increased default to 2000
        max_workers: Number of parallel threads (default: 5)
    """
    all_data = []
    num_batches = math.ceil(total_count / BATCH_SIZE)
    failed_batches = []
    data_lock = Lock()
    
    print(f"\n{'='*70}")
    print(f"Starting PARALLEL batch fetch")
    print(f"Total records: {total_count}")
    print(f"Batch size: {BATCH_SIZE}")
    print(f"Number of batches: {num_batches}")
    print(f"Parallel workers: {max_workers}")
    print(f"Estimated time: ~{(num_batches * 3) / max_workers:.0f}s (vs ~{num_batches * 10:.0f}s sequential)")
    print(f"{'='*70}\n")
    
    def fetch_single_batch(batch_num):
        """Fetch and decrypt a single batch"""
        offset = batch_num * BATCH_SIZE
        current_batch_size = min(BATCH_SIZE, total_count - offset)
        
        try:
            # Fetch encrypted payload
            encrypted_payload = fri_data(key_id, current_batch_size, offset, date, client_id)
            
            # Decrypt
            decrypted_batch = decrypt_data(encrypted_payload)
            
            # Parse
            records = []
            if isinstance(decrypted_batch, str):
                decrypted_json = json.loads(decrypted_batch)
                if isinstance(decrypted_json, dict) and "data" in decrypted_json:
                    records = decrypted_json["data"]
                elif isinstance(decrypted_json, list):
                    records = decrypted_json
            elif isinstance(decrypted_batch, list):
                records = decrypted_batch
            
            return records if records else None
        
        except Exception as e:
            print(f"[Batch-{batch_num+1}] ✗ Error: {e}")
            return None
    
    # Parallel execution
    start_time = time.time()
    
    with concurrent.futures.ThreadPoolExecutor(max_workers=max_workers) as executor:
        future_to_batch = {executor.submit(fetch_single_batch, i): i for i in range(num_batches)}
        
        completed = 0
        for future in concurrent.futures.as_completed(future_to_batch):
            batch_num = future_to_batch[future]
            completed += 1
            
            try:
                records = future.result()
                if records:
                    with data_lock:
                        all_data.extend(records)
                    print(f"✓ Progress: {completed}/{num_batches} ({len(all_data)}/{total_count} records)")
                else:
                    failed_batches.append(batch_num)
                    print(f"⚠ Batch {batch_num+1} failed")
                    
            except Exception as e:
                failed_batches.append(batch_num)
                print(f"✗ Batch {batch_num+1} exception: {e}")
    
    elapsed_time = time.time() - start_time
    
    print(f"\n{'='*70}")
    print(f"✓ Complete in {elapsed_time:.1f}s (avg {elapsed_time/num_batches:.1f}s/batch)")
    print(f"Records: {len(all_data)}/{total_count} ({len(all_data)/total_count*100:.1f}%)")
    if failed_batches:
        print(f"⚠ Failed: {failed_batches}")
    print(f"{'='*70}\n")
    
    return all_data


def save_to_excel(data: list, filename, columns: list):
    """Save data to Excel file"""
    if not data:
        print("No data to save.")
        return
    
    df = pd.DataFrame(data)
    available_columns = [col for col in columns if col in df.columns]
    
    if available_columns:
        df = df[available_columns]
    
    df.to_excel(filename, index=False, engine='openpyxl')
    
    if isinstance(filename, str):
        print(f"✓ Saved {len(df)} records to {filename}")
