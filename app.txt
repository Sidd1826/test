urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

# Token cache
_token_cache = {
    'token': None,
    'expires_at': None
}

# Session with retry logic
_session = None

def get_session_with_retries():
    """Create a requests session with retry logic and connection pooling"""
    global _session
    
    if _session is not None:
        return _session
    
    session = requests.Session()
    
    retry_strategy = Retry(
        total=5,  # Total number of retries
        backoff_factor=2,  # Wait 1, 2, 4, 8, 16 seconds between retries
        status_forcelist=[429, 500, 502, 503, 504],  # Retry on these status codes
        allowed_methods=["POST", "GET"]
    )
    
    adapter = HTTPAdapter(
        max_retries=retry_strategy,
        pool_connections=10,
        pool_maxsize=10,
        pool_block=False
    )
    
    session.mount("http://", adapter)
    session.mount("https://", adapter)
    
    _session = session
    return session


def fri_auth(key_id: str, force_refresh=False) -> str:
    """
    Authenticate with FRI API and cache the token
    
    Args:
        key_id: The encryption key ID
        force_refresh: Force a new token even if cached one is valid
        
    Returns:
        Authentication token string
    """
    global _token_cache
    
    # Return cached token if valid and not forcing refresh
    if not force_refresh and _token_cache['token'] and _token_cache['expires_at']:
        if datetime.now() < _token_cache['expires_at']:
            print("Using cached token")
            return _token_cache['token']
    
    print("Fetching new authentication token...")
    
    payload = {
        "email": os.getenv("FRI_EMAIL"),
        "password": os.getenv("FRI_PASSWORD")
    }
    
    if not payload["email"] or not payload["password"]:
        raise ValueError("FRI_EMAIL and FRI_PASSWORD environment variables must be set")
    
    encrypted_payload = encrypt_data(payload, key_id)
    
    headers = {
        "Content-Type": "application/json"
    }
    
    AUTH_URL = "https://fritestgateway.sancharsaathi.gov.in/fri/v1/auth"
    
    session = get_session_with_retries()
    
    try:
        response = session.post(
            AUTH_URL,
            json=encrypted_payload,
            headers=headers,
            timeout=30,
            verify=False
        )
        
        response.raise_for_status()
        
        if not response.text:
            raise ValueError("Empty response from auth API")
        
        res = response.json()
        token = res.get('token')
        
        if not token:
            raise ValueError(f"No token in response. Response: {res}")
        
        # Cache the token
        _token_cache['token'] = token
        _token_cache['expires_at'] = datetime.now() + timedelta(minutes=120)
        
        print("✓ Authentication successful")
        return token
        
    except requests.exceptions.RequestException as e:
        print(f"✗ Auth API Error: {e}")
        if hasattr(e, 'response') and e.response is not None:
            print(f"Response status: {e.response.status_code}")
            print(f"Response body: {e.response.text}")
        raise
    except json.JSONDecodeError as e:
        print(f"✗ Auth JSON Decode Error: {e}")
        print(f"Response text: {response.text}")
        raise ValueError(f"Invalid JSON response from auth API: {response.text[:500]}")


def fri_count(key_id: str, count_date: str, max_retries=3) -> dict:
    """
    Get count of records for a specific date
    
    Args:
        key_id: The encryption key ID
        count_date: Date in YYYY-MM-DD format
        max_retries: Maximum number of retry attempts
        
    Returns:
        Dictionary with count and date
    """
    session = get_session_with_retries()
    
    for attempt in range(max_retries):
        try:
            token = fri_auth(key_id, force_refresh=(attempt > 0))
            
            if not token:
                raise ValueError("Failed to retrieve FRI auth token")
            
            headers = {
                "Content-Type": "application/json",
                "Authorization": token
            }
            
            body = {"date": count_date}
            FRI_COUNT_URL = "https://fritestgateway.sancharsaathi.gov.in/fri/v1/count"
            
            print(f"Fetching count for date: {count_date} (attempt {attempt + 1}/{max_retries})")
            
            response = session.post(
                FRI_COUNT_URL,
                json=body,
                headers=headers,
                timeout=30,
                verify=False
            )
            
            # Log response details
            print(f"Response Status: {response.status_code}")
            
            # Handle 401 Unauthorized (token expired)
            if response.status_code == 401:
                print("Token expired, refreshing...")
                _token_cache['token'] = None
                _token_cache['expires_at'] = None
                if attempt < max_retries - 1:
                    time.sleep(2)
                    continue
                else:
                    raise ValueError("Authentication failed after retries")
            
            # Handle 429 Too Many Requests (rate limiting)
            if response.status_code == 429:
                retry_after = int(response.headers.get('Retry-After', 10))
                print(f"Rate limited. Waiting {retry_after} seconds...")
                time.sleep(retry_after)
                if attempt < max_retries - 1:
                    continue
                else:
                    raise ValueError("Rate limit exceeded after retries")
            
            response.raise_for_status()
            
            # Check if response has content
            if not response.text or response.text.strip() == '':
                raise ValueError(f"Empty response from count API for date {count_date}")
            
            # Parse JSON
            try:
                result = response.json()
                print(f"✓ Count retrieved: {result.get('count', 0)} records")
                return result
            except json.JSONDecodeError as e:
                print(f"JSON Decode Error: {e}")
                print(f"Response content: {response.text[:500]}")
                raise ValueError(f"Invalid JSON response: {response.text[:500]}")
        
        except requests.exceptions.ConnectionError as e:
            print(f"✗ Connection error on attempt {attempt + 1}: {e}")
            if attempt < max_retries - 1:
                wait_time = (2 ** attempt)  # Exponential backoff
                print(f"Retrying in {wait_time} seconds...")
                time.sleep(wait_time)
            else:
                raise
        
        except requests.exceptions.Timeout as e:
            print(f"✗ Timeout on attempt {attempt + 1}: {e}")
            if attempt < max_retries - 1:
                wait_time = (2 ** attempt)
                print(f"Retrying in {wait_time} seconds...")
                time.sleep(wait_time)
            else:
                raise
        
        except requests.exceptions.RequestException as e:
            print(f"✗ Count API Error: {e}")
            if hasattr(e, 'response') and e.response is not None:
                print(f"Response status: {e.response.status_code}")
                print(f"Response body: {e.response.text}")
            raise
    
    raise ValueError(f"Failed to get count after {max_retries} attempts")


def fri_data(key_id: str, count: int, offset: int, date: str, client_id: str, max_retries=3) -> dict:
    """
    Fetch FRI data with pagination
    
    Args:
        key_id: The encryption key ID
        count: Number of records to fetch
        offset: Starting offset
        date: Date in YYYY-MM-DD format
        client_id: Client identifier
        max_retries: Maximum number of retry attempts
        
    Returns:
        Encrypted payload dictionary
    """
    session = get_session_with_retries()
    
    for attempt in range(max_retries):
        try:
            token = fri_auth(key_id, force_refresh=(attempt > 0))
            
            if not token:
                raise ValueError("Failed to retrieve FRI auth token")
            
            headers = {
                "Content-Type": "application/json",
                "Authorization": token
            }
            
            body = {
                "date": date,
                "client_id": client_id,
                "offset": offset,
                "count": count
            }
            
            FRI_DATA_URL = "https://fritestgateway.sancharsaathi.gov.in/fri/v1/data"
            
            print(f"  Fetching data - offset: {offset}, count: {count} (attempt {attempt + 1}/{max_retries})")
            
            response = session.post(
                FRI_DATA_URL,
                json=body,
                headers=headers,
                timeout=90,  # Increased timeout for data fetch
                verify=False
            )
            
            # Handle 401 Unauthorized
            if response.status_code == 401:
                print("  Token expired, refreshing...")
                _token_cache['token'] = None
                _token_cache['expires_at'] = None
                if attempt < max_retries - 1:
                    time.sleep(2)
                    continue
                else:
                    raise ValueError("Authentication failed after retries")
            
            # Handle 429 Too Many Requests
            if response.status_code == 429:
                retry_after = int(response.headers.get('Retry-After', 10))
                print(f"  Rate limited. Waiting {retry_after} seconds...")
                time.sleep(retry_after)
                if attempt < max_retries - 1:
                    continue
                else:
                    raise ValueError("Rate limit exceeded after retries")
            
            response.raise_for_status()
            
            if not response.text:
                raise ValueError("Empty response from data API")
            
            try:
                result = response.json()
                print(f"  ✓ Data fetched successfully")
                return result
            except json.JSONDecodeError as e:
                print(f"  JSON Decode Error: {e}")
                print(f"  Response content: {response.text[:500]}")
                raise ValueError(f"Invalid JSON response: {response.text[:500]}")
        
        except requests.exceptions.ConnectionError as e:
            print(f"  ✗ Connection error on attempt {attempt + 1}: {e}")
            if attempt < max_retries - 1:
                wait_time = (2 ** attempt) * 2  # Longer backoff for data fetch
                print(f"  Retrying in {wait_time} seconds...")
                time.sleep(wait_time)
                
                # Refresh token on connection error
                _token_cache['token'] = None
                _token_cache['expires_at'] = None
            else:
                print(f"  ✗ Max retries reached for this batch")
                raise
        
        except requests.exceptions.Timeout as e:
            print(f"  ✗ Timeout on attempt {attempt + 1}: {e}")
            if attempt < max_retries - 1:
                wait_time = (2 ** attempt) * 2
                print(f"  Retrying in {wait_time} seconds...")
                time.sleep(wait_time)
            else:
                raise
        
        except requests.exceptions.RequestException as e:
            print(f"  ✗ Data API Error: {e}")
            if hasattr(e, 'response') and e.response is not None:
                print(f"  Response status: {e.response.status_code}")
                print(f"  Response body: {e.response.text[:500]}")
            raise
    
    raise ValueError(f"Failed to fetch data after {max_retries} attempts")


def fetch_and_decrypt_all(key_id: str, total_count: int, date: str, client_id: str, BATCH_SIZE: int = 1000) -> list:
    """
    Fetch and decrypt all data in batches with rate limiting and error handling
    
    Args:
        key_id: The encryption key ID
        total_count: Total number of records to fetch
        date: Date in YYYY-MM-DD format
        client_id: Client identifier
        BATCH_SIZE: Number of records per batch (default: 1000)
        
    Returns:
        List of all decrypted records
    """
    all_data = []
    num_batches = math.ceil(total_count / BATCH_SIZE)
    failed_batches = []
    
    print(f"\n{'='*70}")
    print(f"Starting batch fetch process")
    print(f"Total records: {total_count}")
    print(f"Batch size: {BATCH_SIZE}")
    print(f"Number of batches: {num_batches}")
    print(f"{'='*70}\n")
    
    for batch_num in range(num_batches):
        offset = batch_num * BATCH_SIZE
        current_batch_size = min(BATCH_SIZE, total_count - offset)
        
        print(f"{'─'*70}")
        print(f"Batch {batch_num + 1}/{num_batches}")
        print(f"Offset: {offset}, Size: {current_batch_size}")
        print(f"{'─'*70}")
        
        try:
            # Fetch encrypted payload
            encrypted_payload = fri_data(key_id, current_batch_size, offset, date, client_id)
            
            # Decrypt the batch
            print(f"  Decrypting batch...")
            decrypted_batch = decrypt_data(encrypted_payload)
            
            # Parse records
            records = []
            if isinstance(decrypted_batch, str):
                try:
                    decrypted_json = json.loads(decrypted_batch)
                    
                    if isinstance(decrypted_json, dict) and "data" in decrypted_json:
                        records = decrypted_json["data"]
                    elif isinstance(decrypted_json, list):
                        records = decrypted_json
                    else:
                        print(f"  Warning: Unexpected JSON structure: {type(decrypted_json)}")
                        print(f"  Keys: {list(decrypted_json.keys()) if isinstance(decrypted_json, dict) else 'N/A'}")
                        
                except json.JSONDecodeError as e:
                    print(f"  ✗ Error parsing decrypted batch JSON: {e}")
                    print(f"  First 200 chars: {decrypted_batch[:200]}")
                    failed_batches.append(batch_num)
                    
            elif isinstance(decrypted_batch, list):
                records = decrypted_batch
            else:
                print(f"  Warning: Decrypted batch is unexpected type: {type(decrypted_batch)}")
                failed_batches.append(batch_num)
            
            if records:
                all_data.extend(records)
                print(f"  ✓ Batch {batch_num + 1} complete: {len(records)} records added")
                print(f"  Running total: {len(all_data)}/{total_count} records")
            else:
                print(f"  ⚠ Batch {batch_num + 1} returned no records")
                failed_batches.append(batch_num)
            
            # Rate limiting: delay between batches
            if batch_num < num_batches - 1:  # Don't delay after last batch
                delay = 2  # 2 seconds between batches
                print(f"  Waiting {delay}s before next batch...")
                time.sleep(delay)
        
        except Exception as e:
            print(f"  ✗ Error in batch {batch_num + 1}: {type(e).__name__}: {e}")
            failed_batches.append(batch_num)
            
            # Continue with next batch instead of stopping
            print(f"  Continuing with next batch...")
            time.sleep(5)  # Longer delay after error
            
            # Optionally, uncomment to stop on first error:
            # raise
    
    print(f"\n{'='*70}")
    print(f"Batch fetch process complete")
    print(f"Total records fetched: {len(all_data)}/{total_count}")
    print(f"Success rate: {((num_batches - len(failed_batches)) / num_batches * 100):.1f}%")
    
    if failed_batches:
        print(f"⚠ Failed batches: {failed_batches}")
    else:
        print(f"✓ All batches processed successfully")
    
    print(f"{'='*70}\n")
    
    return all_data


def save_to_excel(data: list, filename, columns: list):
    """
    Save data to Excel file
    
    Args:
        data: List of records
        filename: Output file path or BytesIO object
        columns: List of column names to include
    """
    if not data:
        print("No data to save.")
        return
    
    df = pd.DataFrame(data)
    
    # Reorder and filter columns if they exist
    available_columns = [col for col in columns if col in df.columns]
    if available_columns:
        df = df[available_columns]
    else:
        print(f"Warning: None of the specified columns {columns} found in data")
        print(f"Available columns: {list(df.columns)}")
    
    df.to_excel(filename, index=False, engine='openpyxl')
    
    if isinstance(filename, str):
        print(f"✓ Data saved to {filename}")
        print(f"  Columns: {list(df.columns)}")
        print(f"  Rows: {len(df)}")
