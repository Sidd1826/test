Blob structure:

def _get_container_name(module: str) -> str:
    """Map module string to its blob container name."""
    return AZURE_CONTAINER_FRI if module.upper() == "FRI" else AZURE_CONTAINER_MNRL


def _pg_conn():
    """Open and return a new psycopg2 connection (used for ALL DB operations)."""
    return psycopg2.connect(PG_DSN)

    """Return a BlobServiceClient from the configured connection string."""
    return BlobServiceClient.from_connection_string(AZURE_STORAGE_CONNECTION_STRING)

def build_excel(records: list, columns: list) -> io.BytesIO:
    """
    Build an .xlsx file from records using openpyxl write-only mode.
    Rows are flushed incrementally — peak RAM stays flat regardless of
    how many records are passed (tested up to 10-lakh rows).

    Args:
        records : list of dicts, each keyed by column name
        columns : ordered list of column keys to include

    Returns:
        io.BytesIO positioned at 0, ready to read or upload
    """
    buf = io.BytesIO()
    wb  = Workbook(write_only=True)
    ws  = wb.create_sheet("Data")

    # ── Header row ──
    header_font  = Font(bold=True, color="FFFFFF")
    header_fill  = PatternFill("solid", fgColor="1D4ED8")
    header_align = Alignment(horizontal="center")

    header_cells = []
    for col in columns:
        cell           = WriteOnlyCell(ws, value=col.replace("_", " ").title())
        cell.font      = header_font
        cell.fill      = header_fill
        cell.alignment = header_align
        header_cells.append(cell)
    ws.append(header_cells)

    # ── Data rows written in 5k-row chunks ──
    CHUNK = 5_000
    for start in range(0, len(records), CHUNK):
        for rec in records[start : start + CHUNK]:
            ws.append([rec.get(c, "") for c in columns])

    wb.save(buf)
    buf.seek(0)
    return buf


def build_matched_excel(
    input_rows: list[dict],
    headers: list[str],
    mob_col: str,
    matched_map: dict[str, str],
) -> tuple[io.BytesIO, int]:
    """
    Write only the matched rows to a new Excel file.

    Output columns = all original Excel columns + cust_id (appended last).
    Uses openpyxl write-only mode — RAM stays flat regardless of row count.

    Args:
        input_rows  : all rows read from the source Excel
        headers     : original column order from the source Excel
        mob_col     : name of the mobile-number column in the Excel
        matched_map : { final_mob_num: cust_id } from mobilecleanup

    Returns:
        (BytesIO ready to upload, matched_row_count)
    """
    output_headers = headers + ["cust_id"]

    buf = io.BytesIO()
    wb  = Workbook(write_only=True)
    ws  = wb.create_sheet("Matched Data")

    # ── Header row ──
    header_font  = Font(bold=True, color="FFFFFF")
    header_fill  = PatternFill("solid", fgColor="1D4ED8")
    header_align = Alignment(horizontal="center")

    header_cells = []
    for col_name in output_headers:
        cell           = WriteOnlyCell(ws, value=col_name.replace("_", " ").title())
        cell.font      = header_font
        cell.fill      = header_fill
        cell.alignment = header_align
        header_cells.append(cell)
    ws.append(header_cells)

    # ── Data rows — flushed in batches to keep RAM flat ──
    matched_count = 0
    buffer: list  = []

    for row in input_rows:
        mob_val = str(row.get(mob_col, "")).strip()
        if mob_val not in matched_map:
            continue

        data_row = [row.get(h, "") for h in headers]
        data_row.append(matched_map[mob_val])   # cust_id appended last
        buffer.append(data_row)
        matched_count += 1

        if len(buffer) >= EXCEL_FLUSH_ROWS:
            for r in buffer:
                ws.append(r)
            buffer.clear()

    for r in buffer:   # flush remainder
        ws.append(r)

    wb.save(buf)
    buf.seek(0)
    return buf, matched_count

def upload_blob(container: str, blob_name: str, data: bytes) -> str:
    """
    Upload raw bytes to Azure Blob Storage.
    Returns the canonical blob URL (no SAS token).
    """
    blob_client = _blob_service().get_blob_client(container=container, blob=blob_name)
    blob_client.upload_blob(data, overwrite=True)
    return blob_client.url


def get_sas_download_url(container: str, blob_name: str, expiry_hours: int = 1) -> str:
    """
    Generate a short-lived read-only SAS URL so the browser can download
    directly from Azure Blob — no server proxy required.

    Returns a full HTTPS URL with embedded SAS token.
    """
    sas_token = generate_blob_sas(
        account_name=AZURE_STORAGE_ACCOUNT_NAME,
        container_name=container,
        blob_name=blob_name,
        account_key=AZURE_STORAGE_ACCOUNT_KEY,
        permission=BlobSasPermissions(read=True),
        expiry=datetime.now(timezone.utc) + timedelta(hours=expiry_hours),
    )
    return (
        f"https://{AZURE_STORAGE_ACCOUNT_NAME}.blob.core.windows.net"
        f"/{container}/{blob_name}?{sas_token}"
    )

def list_source_date_folders(container: str) -> list[str]:
    """
    Return all date-folder prefixes inside a source container,
    sorted newest first.
    e.g. ["2025-03-12", "2025-03-11", "2025-03-10"]

    Scans blob names and extracts the first path segment.
    Works even if Azure doesn't store explicit "folder" blobs.

    Raises:
        ValueError — if container is not a permitted source
    """
    if container not in VALID_SOURCE_CONTAINERS.values():
        raise ValueError(f"Container '{container}' is not a permitted source.")

    cc   = _blob_service().get_container_client(container)
    seen = set()
    for blob in cc.list_blobs():
        parts = blob.name.split("/")
        if len(parts) >= 2 and parts[0]:
            seen.add(parts[0])

    return sorted(seen, reverse=True)


def list_source_files(container: str, date_folder: str) -> list[str]:
    """
    Return all .xlsx filenames inside <container>/<date_folder>/.
    e.g. ["mnrl_normal_2025-03-10_50000_records.xlsx",
           "mnrl_reactivated_2025-03-10_12000_records.xlsx"]

    Returns only the filename part (not the full blob path).

    Raises:
        ValueError — if container is not a permitted source
    """
    if container not in VALID_SOURCE_CONTAINERS.values():
        raise ValueError(f"Container '{container}' is not a permitted source.")

    prefix = f"{date_folder}/"
    cc     = _blob_service().get_container_client(container)
    files  = []
    for blob in cc.list_blobs(name_starts_with=prefix):
        fname = blob.name[len(prefix):]
        if fname and fname.endswith(".xlsx") and "/" not in fname:
            files.append(fname)

    return sorted(files)


# ══════════════════════════════════════════════════════════════
# BLOB READ — source Excel files
# ══════════════════════════════════════════════════════════════

def read_excel_from_blob(
    container: str,
    date_folder: str,
    file_name: str,
) -> tuple[list[dict], list[str]]:
    """
    Stream an Excel file from <container>/<date_folder>/<file_name>
    directly into memory. No temp file, no disk I/O.

    Returns:
        rows    : list of dicts  { column_header: cell_value }
        headers : ordered list of column names (original order preserved)

    Raises:
        ValueError — blob not found, empty file, or no header row
    """
    if container not in VALID_SOURCE_CONTAINERS.values():
        raise ValueError(f"Container '{container}' is not a permitted source.")

    blob_path = f"{date_folder}/{file_name}"
    bc        = _blob_service().get_blob_client(container=container, blob=blob_path)

    try:
        stream = io.BytesIO()
        bc.download_blob().readinto(stream)
        stream.seek(0)
    except Exception as e:
        raise ValueError(
            f"Could not read blob '{blob_path}' from '{container}': {e}"
        ) from e

    # read_only=True keeps RAM flat on large files
    wb        = load_workbook(filename=stream, read_only=True, data_only=True)
    ws        = wb.active
    rows_iter = ws.iter_rows(values_only=True)

    try:
        raw_headers = next(rows_iter)
    except StopIteration:
        raise ValueError(f"Excel file '{file_name}' is empty.")

    headers = [
        str(h).strip() if h is not None else f"col_{i}"
        for i, h in enumerate(raw_headers)
    ]

    rows = [dict(zip(headers, row)) for row in rows_iter]
    wb.close()

    if not rows:
        raise ValueError(f"Excel file '{file_name}' has a header row but no data rows.")

    return rows, headers

# ══════════════════════════════════════════════════════════════
# MOBILECLEANUP — chunked parallel matching (pure psycopg2)
#
# Phase 2 TRUNCATES + bulk-loads this table daily at 7 AM.
# Phase 3 matches Excel mob_num against final_mob_num here.
# All queries use psycopg2 — no pyodbc, no Azure SQL.
# ══════════════════════════════════════════════════════════════

def _fetch_chunk(mob_chunk: list[str]) -> dict[str, str]:
    """
    Query one chunk of mob numbers from the mobilecleanup table.
    Returns { final_mob_num: cust_id } for matched rows only.

    Key detail — psycopg2 IN %s syntax:
        cur.execute("... WHERE col IN %s", (tuple(values),))
        The outer tuple is psycopg2's parameter wrapper.
        The inner tuple is the actual IN list.
        This is different from pyodbc which uses IN (?,?,?).

    Opens its own connection — safe to run inside a thread.
    """
    if not mob_chunk:
        return {}

    sql = f"""
        SELECT {PG_MOB_COLUMN}, {PG_CUSTID_COLUMN}
        FROM   {PG_MATCHOFF_TABLE}
        WHERE  {PG_MOB_COLUMN} IN %s
    """

    result = {}
    with _pg_conn() as conn, conn.cursor() as cur:
        cur.execute(sql, (tuple(mob_chunk),))
        for row in cur.fetchall():
            mob_val  = str(row[0]).strip()
            cust_val = str(row[1]).strip() if row[1] is not None else ""
            result[mob_val] = cust_val

    return result


def match_mob_numbers(
    mob_numbers: list,
    chunk_size: int = SQL_CHUNK_SIZE,
    max_workers: int = SQL_MAX_WORKERS,
) -> dict[str, str]:
    """
    Match a list of mobile numbers against mobilecleanup in parallel chunks.

    Flow:
      1. Normalise to strings + deduplicate   → reduces query load on 30L rows
      2. Split into chunks of `chunk_size`    → keeps each IN clause manageable
      3. Fire chunks via ThreadPoolExecutor   → parallel I/O, flat CPU
      4. Merge all results                    → single { final_mob_num: cust_id }

    Returns:
        { matched_final_mob_num: cust_id, ... }
    """
    unique_mobs = list({
        str(m).strip()
        for m in mob_numbers
        if m is not None and str(m).strip()
    })

    if not unique_mobs:
        return {}

    chunks = [
        unique_mobs[i : i + chunk_size]
        for i in range(0, len(unique_mobs), chunk_size)
    ]

    matched: dict[str, str] = {}

    with ThreadPoolExecutor(max_workers=max_workers) as pool:
        futures = {pool.submit(_fetch_chunk, chunk): chunk for chunk in chunks}
        for future in as_completed(futures):
            try:
                matched.update(future.result())
            except Exception as exc:
                # Log but continue — partial results better than a total abort
                print(f"[MATCHOFF] Chunk query failed: {exc}")

    return matched


# ══════════════════════════════════════════════════════════════
# POSTGRESQL — export_files CRUD
# ══════════════════════════════════════════════════════════════

def save_export_metadata(
    module: str,
    file_name: str,
    blob_name: str,
    record_count: int,
    export_date: str,
    data_type: str = None,
) -> str:
    """
    Persist export metadata to PostgreSQL after a successful Blob upload.
    Returns the UUID string of the newly created row.
    """
    row_id    = str(uuid.uuid4())
    container = _get_container_name(module)

    with _pg_conn() as conn, conn.cursor() as cur:
        cur.execute(
            """
            INSERT INTO export_files
                (id, module, file_name, blob_name, container,
                 record_count, export_date, data_type, created_at)
            VALUES (%s, %s, %s, %s, %s, %s, %s, %s, NOW())
            """,
            (
                row_id, module.upper(), file_name, blob_name, container,
                record_count, export_date, data_type,
            ),
        )
    return row_id


def list_export_metadata(module: str) -> list:
    """
    Fetch all export rows for a module from PostgreSQL, newest first.
    Returns a list of plain dicts (JSON-serialisable).
    """
    with _pg_conn() as conn, conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor) as cur:
        cur.execute(
            """
            SELECT id, file_name, blob_name, container,
                   record_count, export_date, data_type, created_at
            FROM   export_files
            WHERE  module = %s
            ORDER  BY created_at DESC
            """,
            (module.upper(),),
        )
        rows = cur.fetchall()

    result = []
    for row in rows:
        d = dict(row)
        d["created_at"]  = d["created_at"].isoformat()
        d["export_date"] = str(d["export_date"])
        result.append(d)
    return result


def get_export_metadata_row(file_id: str, module: str) -> dict:
    """
    Fetch a single export metadata row by UUID + module.
    Raises ValueError if not found.
    """
    with _pg_conn() as conn, conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor) as cur:
        cur.execute(
            "SELECT * FROM export_files WHERE id = %s AND module = %s",
            (file_id, module.upper()),
        )
        row = cur.fetchone()

    if not row:
        raise ValueError(f"No export record found: id={file_id}, module={module}")
    return dict(row)

def save_matchoff_metadata(
    source_container: str,
    source_date_folder: str,
    source_file_name: str,
    output_blob_path: str,
    input_row_count: int,
    matched_count: int,
    mob_col: str,
    run_date: str,
) -> str:
    """Persist a matchoff run to PostgreSQL. Returns new row UUID."""
    row_id = str(uuid.uuid4())

    with _pg_conn() as conn, conn.cursor() as cur:
        cur.execute(
            """
            INSERT INTO matchoff_runs (
                id,
                source_container,   source_date_folder, source_file_name,
                output_blob_path,   output_container,
                input_row_count,    matched_count,
                mob_column,         run_date,
                created_at
            )
            VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, NOW())
            """,
            (
                row_id,
                source_container,   source_date_folder, source_file_name,
                output_blob_path,   AZURE_CONTAINER_MATCHOFF_OUTPUT,
                input_row_count,    matched_count,
                mob_col,            run_date,
            ),
        )
    return row_id


def list_matchoff_metadata() -> list:
    """All matchoff runs from PostgreSQL, newest first."""
    with _pg_conn() as conn, conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor) as cur:
        cur.execute(
            """
            SELECT id,
                   source_container,   source_date_folder, source_file_name,
                   output_blob_path,   output_container,
                   input_row_count,    matched_count,
                   mob_column,         run_date,
                   created_at
            FROM   matchoff_runs
            ORDER  BY created_at DESC
            """
        )
        rows = cur.fetchall()

    result = []
    for row in rows:
        d = dict(row)
        d["created_at"] = d["created_at"].isoformat()
        d["run_date"]   = str(d["run_date"])
        result.append(d)
    return result


def get_matchoff_run(run_id: str) -> dict:
    """Single matchoff row by UUID. Raises ValueError if not found."""
    with _pg_conn() as conn, conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor) as cur:
        cur.execute("SELECT * FROM matchoff_runs WHERE id = %s", (run_id,))
        row = cur.fetchone()
    if not row:
        raise ValueError(f"Matchoff run not found: {run_id}")
    return dict(row)

def export_and_store(
    module: str,
    records: list,
    columns: list,
    file_name: str,
    export_date: str,
    data_type: str = None,
) -> dict:
    """
    Build Excel → upload to Azure Blob → save metadata to PostgreSQL.

    Args:
        module      : "FRI" or "MNRL"
        records     : list of dicts from the decryption layer
        columns     : ordered column keys for the Excel sheet
        file_name   : desired filename
        export_date : ISO date string "YYYY-MM-DD"
        data_type   : "Normal" | "Reactivated" | None

    Returns:
        dict with: id, file_name, record_count, export_date, module, data_type
    """
    container = _get_container_name(module)
    blob_name = f"{export_date}/{file_name}"

    buf = build_excel(records, columns)
    upload_blob(container, blob_name, buf.read())

    row_id = save_export_metadata(
        module=module,
        file_name=file_name,
        blob_name=blob_name,
        record_count=len(records),
        export_date=export_date,
        data_type=data_type,
    )

    return {
        "id":           row_id,
        "file_name":    file_name,
        "record_count": len(records),
        "export_date":  export_date,
        "module":       module.upper(),
        "data_type":    data_type,
    }


def run_matchoff(
    source_container: str,
    date_folder: str,
    file_name: str,
    mob_col: str = "mob_num",
) -> dict:
    """
    Full Phase 3 matchoff pipeline.

    Steps:
      1. Read source Excel directly from Blob (no disk I/O)
      2. Extract + deduplicate mob_num column
      3. Chunked parallel query against mobilecleanup (final_mob_num → cust_id)
      4. Build matched output Excel (all original columns + cust_id)
      5. Upload result to matchoff-output/<date_folder>/
      6. Save run metadata to PostgreSQL
      7. Return 24-hour SAS download URL

    Args:
        source_container : "fri-exports" or "mnrl-exports"
        date_folder      : "2025-03-10"
        file_name        : "mnrl_normal_2025-03-10_50000_records.xlsx"
        mob_col          : column header in the Excel holding the mobile number
                           (matched against final_mob_num in mobilecleanup)

    Returns dict (becomes job.result on the frontend):
        {
          "id":               <uuid>,
          "source_container": "mnrl-exports",
          "date_folder":      "2025-03-10",
          "source_file":      "mnrl_normal_..._records.xlsx",
          "output_file":      "mnrl_normal_..._matched_1234_records.xlsx",
          "input_row_count":  50000,
          "matched_count":    1234,
          "sas_url":          "https://...?sv=...&sig=...",
          "run_date":         "2025-03-10",
        }
    """
    run_date = datetime.now(timezone.utc).strftime("%Y-%m-%d")

    # ── 1. Read source Excel from existing blob ──
    print(f"[MATCHOFF] Reading {source_container}/{date_folder}/{file_name}")
    input_rows, headers = read_excel_from_blob(source_container, date_folder, file_name)
    print(f"[MATCHOFF] {len(input_rows)} rows, columns: {headers}")

    if mob_col not in headers:
        raise ValueError(
            f"Column '{mob_col}' not found in '{file_name}'. "
            f"Available columns: {', '.join(headers)}"
        )

    # ── 2. Extract + deduplicate mob numbers ──
    mob_numbers  = [row.get(mob_col) for row in input_rows]
    unique_count = len({str(m).strip() for m in mob_numbers if m})
    print(f"[MATCHOFF] {unique_count} unique mob numbers to match against mobilecleanup")

    # ── 3. Chunked parallel PostgreSQL match ──
    print(
        f"[MATCHOFF] Querying mobilecleanup in chunks of {SQL_CHUNK_SIZE} "
        f"with {SQL_MAX_WORKERS} parallel workers"
    )
    matched_map = match_mob_numbers(mob_numbers)
    print(f"[MATCHOFF] {len(matched_map)} unique numbers matched")

    if not matched_map:
        raise ValueError(
            "Zero matches found. Check that the mob_col name is correct "
            "and Phase 2 data has been loaded into mobilecleanup."
        )

    # ── 4. Build output Excel ──
    base_name     = file_name.replace(".xlsx", "")
    preview_count = sum(
        1 for row in input_rows
        if str(row.get(mob_col, "")).strip() in matched_map
    )
    output_file_name = f"{base_name}_matched_{preview_count}_records.xlsx"

    print(f"[MATCHOFF] Building output Excel: {output_file_name}")
    buf, matched_count = build_matched_excel(input_rows, headers, mob_col, matched_map)

    # ── 5. Upload to matchoff-output/<date_folder>/ ──
    output_blob_path = upload_matchoff_output(buf, date_folder, output_file_name)
    print(f"[MATCHOFF] Uploaded → matchoff-output/{output_blob_path}")

    # ── 6. Save metadata to PostgreSQL ──
    run_id = save_matchoff_metadata(
        source_container   = source_container,
        source_date_folder = date_folder,
        source_file_name   = file_name,
        output_blob_path   = output_blob_path,
        input_row_count    = len(input_rows),
        matched_count      = matched_count,
        mob_col            = mob_col,
        run_date           = run_date,
    )

    # ── 7. 24-hour SAS URL for immediate download ──
    sas_url = get_sas_download_url(
        container    = AZURE_CONTAINER_MATCHOFF_OUTPUT,
        blob_name    = output_blob_path,
        expiry_hours = 24,
    )

    print(f"[MATCHOFF] Done. matched_count={matched_count}, run_id={run_id}")

    return {
        "id":               run_id,
        "source_container": source_container,
        "date_folder":      date_folder,
        "source_file":      file_name,
        "output_file":      output_file_name,
        "input_row_count":  len(input_rows),
        "matched_count":    matched_count,
        "sas_url":          sas_url,
        "run_date":         run_date,
    }
