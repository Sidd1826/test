from concurrent.futures import ThreadPoolExecutor, as_completed
import io, csv

def run_pipeline() -> None:
    log.info("=" * 60)
    log.info("Pipeline START")
    log.info("BQ Table : %s", BQ_TABLE)
    log.info("Filter   : %s", PARTITION_FILTER)
    log.info("SQL Table: %s", SQL_TABLE)
    log.info("=" * 60)

    total_rows = get_total_count()
    if total_rows == 0:
        log.warning("No rows found – pipeline exiting.")
        return

    # ── Fix 1: compute exact batch sizes so last batch doesn't over-fetch ──
    offsets = []
    for b in range(0, total_rows, BATCH_SIZE):
        actual_size = min(BATCH_SIZE, total_rows - b)  # cap last batch
        offsets.append((b, actual_size))

    num_batches = len(offsets)
    log.info(
        "Total rows=%d  batches=%d  batch_size=%d  last_batch_size=%d",
        total_rows, num_batches, BATCH_SIZE, offsets[-1][1],
    )

    sql_engine = _get_sql_engine()
    truncate_table(sql_engine)

    total_fetched  = 0
    total_skipped  = 0
    total_inserted = 0
    pipeline_start = time.time()

    MAX_WORKERS = int(os.environ.get("PIPELINE_WORKERS", 4))

    try:
        with ThreadPoolExecutor(max_workers=MAX_WORKERS) as executor:
            # ── Fix 1 cont: pass actual_size per batch, not global BATCH_SIZE ──
            futures = {
                executor.submit(fetch_and_clean_batch, offset, size): (batch_num, offset, size)
                for batch_num, (offset, size) in enumerate(offsets, 1)
            }

            for future in as_completed(futures):
                batch_num, offset, size = futures[future]
                batch_start = time.time()

                # ── Fix 2: structured log block per batch ──────────────────
                log.info("─" * 60)
                log.info("BATCH %d/%d  (offset=%d  expected=%d rows)",
                         batch_num, num_batches, offset, size)

                try:
                    insert_rows, fetched, skipped = future.result()
                except Exception as exc:
                    log.error("  [FETCH+CLEAN] FAILED: %s", exc)
                    continue

                log.info("  [FETCH]  rows fetched  : %d", fetched)
                log.info("  [CLEAN]  valid         : %d", fetched - skipped)
                log.info("  [CLEAN]  skipped       : %d  (invalid/empty mobile)", skipped)

                if insert_rows:
                    log.info("  [INSERT] starting COPY of %d rows ...", len(insert_rows))
                    insert_start = time.time()
                    inserted     = bulk_insert_rows(sql_engine, insert_rows)
                    log.info("  [INSERT] done — %d rows in %.2fs",
                             inserted, time.time() - insert_start)
                else:
                    inserted = 0
                    log.info("  [INSERT] skipped — no valid rows")

                total_fetched  += fetched
                total_skipped  += skipped
                total_inserted += inserted

                log.info("  [BATCH]  time          : %.2fs", time.time() - batch_start)
                log.info("  [TOTAL]  cumulative inserted: %d / ~%d",
                         total_inserted, total_rows)

    finally:
        sql_engine.dispose()
        log.info("=" * 60)
        log.info("SQL engine disposed.")

    total_elapsed = time.time() - pipeline_start
    log.info("=" * 60)
    log.info("Pipeline COMPLETE")
    log.info("  Total rows in BQ   : %d", total_rows)
    log.info("  Total fetched      : %d", total_fetched)
    log.info("  Total skipped      : %d  (%.1f%%)",
             total_skipped, (total_skipped / total_fetched * 100) if total_fetched else 0)
    log.info("  Total inserted     : %d  (%.1f%%)",
             total_inserted, (total_inserted / total_fetched * 100) if total_fetched else 0)
    log.info("  Time elapsed       : %.2fs  (%.1f min)",
             total_elapsed, total_elapsed / 60)
    log.info("=" * 60)
