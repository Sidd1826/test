pipeline_start = time.time()

    try:
        for batch_num in range(num_batches):
            batch_start = time.time()
            offset      = batch_num * BATCH_SIZE

            raw_rows = fetch_batch_with_retry(offset, BATCH_SIZE)
            if not raw_rows:
                log.info("Empty batch at offset %d – stopping early.", offset)
                break

            total_fetched   += len(raw_rows)
            has_country_col  = "COUNTRY_RES" in raw_rows[0]

            batch_inserted = 0
            batch_skipped  = 0

            # ── process and insert in SQL_INSERT_CHUNK sub-chunks ──────────
            for sub_chunk in _chunked(raw_rows, SQL_INSERT_CHUNK):
                insert_rows: List[Tuple[str, str, str]] = []

                for row in sub_chunk:
                    cust_id     = str(row.get(CUST_ID_COL, "") or "")
                    mob_num     = str(row.get(DISTINCT_KEY_COL, "") or "")
                    country_res = (
                        str(row.get("COUNTRY_RES", "") or DEFAULT_COUNTRY_RES)
                        if has_country_col
                        else DEFAULT_COUNTRY_RES
                    )
                    final_mob_num = clean_mobile_number(mob_num, country_res)

                    if not final_mob_num:
                        batch_skipped += 1
                        continue

                    insert_rows.append((cust_id, mob_num, final_mob_num))

                if insert_rows:
                    inserted        = bulk_insert_rows(sql_engine, insert_rows)
                    batch_inserted += inserted

                log.info(
                    "  Sub-chunk done — cleaned=%d  skipped=%d  inserted=%d",
                    len(insert_rows),
                    batch_skipped,
                    batch_inserted,
                )

            total_skipped  += batch_skipped
            total_inserted += batch_inserted
            batch_elapsed   = time.time() - batch_start

            log.info(
                "Batch %d/%d  fetched=%d  skipped=%d  inserted=%d  "
                "cumulative=%d  batch_time=%.2fs",
                batch_num + 1,
                num_batches,
                len(raw_rows),
                batch_skipped,
                batch_inserted,
                total_inserted,
                batch_elapsed,
            )
