@routes.route("/mnrl/data", methods=["POST"])
def data_mnrl():
    try:
        key_id = "DIU"
        if not key_id:
            return {"error": "KEY-ID is required"}, 400
        
        req = request.json
        
        if not req.get("date"):
            return {"error": "date is required"}, 400
        
        # Get count first
        count_res = mnrl_count(
            key_id=key_id,
            count_date=req.get("date")
        )
        
        total_count = count_res.get("count", 0)
        print(f"MNRL count: {total_count}")
        
        if total_count == 0:
            return {"error": "No records found for this date", "count": 0}, 404
        
        response = mnrl_data(
            key_id=key_id,
            total_count=total_count,
            date=req.get("date"),
            BATCH_SIZE=req.get("batch_size", 2000),  # Allow custom batch size
            Bank_Id=req.get("BankId"),
            max_workers=req.get("max_workers", 5)  # Allow custom workers
        )
        
        return {"data": response, "total_records": len(response)}, 200
    except Exception as e:
        print(traceback.format_exc())
        return {"error": str(e)}, 500


@routes.route("/mnrl/data/excel", methods=["POST"])
def data_excel_mnrl():
    try:
        key_id = "DIU"
        if not key_id:
            return {"error": "KEY-ID is required"}, 400
        
        req = request.json
        
        if not req.get("date"):
            return {"error": "date is required"}, 400
        
        print(f"\n{'='*70}")
        print(f"MNRL Excel Export Request")
        print(f"Date: {req.get('date')}")
        print(f"{'='*70}\n")
        
        # Get count
        count_res = mnrl_count(
            key_id=key_id,
            count_date=req.get("date")
        )
        
        total_count = count_res.get("count", 0)
        print(f"Total MNRL count: {total_count}")
        
        if total_count == 0:
            return {"error": "No records found for this date", "count": 0}, 404
        
        # Fetch and decrypt all data
        response = mnrl_data(
            key_id=key_id,
            total_count=total_count,
            date=req.get("date"),
            BATCH_SIZE=req.get("batch_size", 2000),
            Bank_Id=req.get("BankId"),
            max_workers=req.get("max_workers", 5)
        )
        
        if not response:
            return {"error": "No data retrieved", "count": 0}, 404
        
        # Prepare Excel file
        columns = ['date_of_disconnection', 'disconnection_reason_id', 'lsa_id', 'mobile_no', 'tsp_id']
        output = io.BytesIO()
        
        print(f"\nGenerating Excel file with {len(response)} records...")
        save_to_excel(response, output, columns)
        output.seek(0)
        
        filename = f"mnrl_data_{req.get('date')}_{len(response)}_records.xlsx"
        
        print(f"✓ Excel file ready: {filename}\n")
        
        return send_file(
            output,
            mimetype="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
            as_attachment=True,
            download_name=filename
        )
        
    except Exception as e:
        print(f"\n✗ Error in /mnrl/data/excel:")
        print(traceback.format_exc())
        return {"error": str(e)}, 500
