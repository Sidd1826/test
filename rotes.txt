@routes.route("/encrypt", methods=["POST"])
def encrypt():
    try:
        key_id = request.headers.get("X-PUBLIC-KEY-ID")
        if not key_id:
            return {"error": "X-PUBLIC-KEY-ID header required"}, 400
        
        payload = request.json
        result = encrypt_data(payload, key_id)
        
        return jsonify(result), 200
    except Exception as e:
        print(traceback.format_exc())
        return {"error": str(e)}, 500


@routes.route("/decrypt", methods=["POST"])

@routes.route("/fri/data", methods=["POST"])
def data():
    try:
        key_id = "FRI"
        if not key_id:
            return {"error": "KEY-ID is required"}, 400
        
        req = request.json
        
        # Validate required fields
        if not req.get("date"):
            return {"error": "date is required"}, 400
        if not req.get("client_id"):
            return {"error": "client_id is required"}, 400
        
        # Get count first
        count_res = fri_count(key_id=key_id, count_date=req.get("date"))
        total_count = count_res.get("count", 0)
        
        if total_count == 0:
            return {"error": "No records found for this date", "count": 0}, 404
        
        response = fetch_and_decrypt_all(
            key_id=key_id,
            total_count=total_count,
            date=req.get("date"),
            client_id=req.get("client_id"),
            BATCH_SIZE=req.get("batch_size", 1000)  # Allow custom batch size
        )
        
        return {"data": response, "total_records": len(response)}, 200
    except Exception as e:
        print(traceback.format_exc())
        return {"error": str(e)}, 500


@routes.route("/fri/data/excel", methods=["POST"])
def data_excel():
    try:
        key_id = "FRI"
        if not key_id:
            return {"error": "KEY-ID is required"}, 400
        
        req = request.json
        
        # Validate required fields
        if not req.get("date"):
            return {"error": "date is required"}, 400
        
        client_id = req.get("client_id", "123412114")  # Default client ID
        batch_size = req.get("batch_size", 1000)  # Default batch size
        
        print(f"\n{'='*70}")
        print(f"Excel Export Request")
        print(f"Date: {req.get('date')}")
        print(f"Client ID: {client_id}")
        print(f"Batch Size: {batch_size}")
        print(f"{'='*70}\n")
        
        # Get count
        count_res = fri_count(
            key_id=key_id,
            count_date=req.get("date")
        )
        
        total_count = count_res.get("count", 0)
        print(f"Total count from API: {total_count}")
        
        if total_count == 0:
            return {"error": "No records found for this date", "count": 0}, 404
        
        # Fetch and decrypt all data
        response = fetch_and_decrypt_all(
            key_id=key_id,
            total_count=total_count,
            date=req.get("date"),
            client_id=client_id,
            BATCH_SIZE=batch_size
        )
        
        if not response:
            return {"error": "No data retrieved", "count": 0}, 404
        
        # Prepare Excel file
        columns = ['fri', 'isa_id', 'mobile_no', 'status', 'tsp_id']
        output = io.BytesIO()
        
        print(f"\nGenerating Excel file with {len(response)} records...")
        save_to_excel(response, output, columns)
        output.seek(0)
        
        filename = f"fri_data_{req.get('date')}_{len(response)}_records.xlsx"
        
        print(f"✓ Excel file ready: {filename}\n")
        
        return send_file(
            output,
            mimetype="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
            as_attachment=True,
            download_name=filename
        )
        
    except Exception as e:
        print(f"\n✗ Error in /fri/data/excel:")
        print(traceback.format_exc())
        return {"error": str(e)}, 500


@routes.route("/dataverse/save", methods=["POST"])
def upload_file():
    try:
        file = request.files['file']  # FileStorage object
        record_count = request.form['record_count']
        export_date = request.form['export_date']
        form_type = request.form['type'].upper()  # MNRL or FRI
        
        print(f"Upload request - Type: {form_type}, Records: {record_count}, Date: {export_date}")
        
        # Read file contents
        contents = file.read()
        filename = file.filename
        
        row = {
            'cr1d9_recordcount': record_count,
            'cr1d9_exportdate': export_date
        }
        
        if form_type == 'FRI':
            if not record_count or not export_date:
                return {"error": "Missing FRI fields"}, 400
            table_name = 'cr1d9_fri_downloads'
            
        elif form_type == 'MNRL':
            data_type = request.form.get('data_type')
            if not record_count or not export_date or not data_type:
                return {"error": "Missing MNRL fields"}, 400
            row['cr1d9_datatype'] = data_type
            table_name = 'cr1d9_mnrl_downloads'
        else:
            return {"error": f"Invalid type: {form_type}"}, 400
        
        # Call backend logic
        result = create_row_file(
            table_name=table_name,
            row=row,
            file_col_name='cr1d9_file',
            filename=filename,
            contents=contents
        )
        
        return jsonify({'success': True, 'result': result})
        
    except Exception as e:
        print(traceback.format_exc())
        return {"error": str(e)}, 500

@routes.route("/dataverse/download/<fileId>", methods=["GET"])
def download(fileId):
    try:
        req_type = request.args.get('type', 'FRI').upper()
        
        if req_type == 'FRI':
            table_name = 'cr1d9_fri_downloads'
        elif req_type == 'MNRL':
            table_name = 'cr1d9_mnrl_downloads'
        else:
            return {"error": f"Invalid type: {req_type}"}, 400
        
        data, fileName = download_file(
            table_name=table_name,
            row_id=fileId,
            file_col_name="cr1d9_file"
        )
        
        return Response(
            data,
            mimetype="application/octet-stream",
            headers={
                "Content-Disposition": f'attachment; filename="{fileName}"'
            }
        )
        
    except Exception as e:
        print(traceback.format_exc())
        return {"error": str(e)}, 500
